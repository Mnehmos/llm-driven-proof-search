    have hsub : n - 6 < n := by omega

      refine ⟨2 * n, ?_⟩
      rintro y ⟨x, rfl⟩
      dsimp [f]
      rw [ArithmeticFunction.sigma_zero_apply]
      have hc := Nat.card_divisors_le_self (x : ℕ)
      have hx : (x : ℕ) < n := x.isLt
      omega
    let m : Fin n := ⟨n - 6, hsub⟩
    have hm : f m ≤ n + 2 := le_trans (le_ciSup hbdd m) H
    dsimp [f, m] at hm
    omega
  rcases h2520 with ⟨q, hq⟩
  have h6div : 6 ∣ n - 6 := by
    apply Nat.dvd_iff_mod_eq_zero.mpr
    omega
  let r : ℕ := (n - 6) / 6
  have hcancel := Nat.div_mul_cancel h6div
  have hr : n - 6 = 6 * r := by
    dsimp [r]
    omega
  change Nat.Prime r
  have hr2 : 2 ≤ r := by omega
  have hrmod : r % 6 = 5 := by omega
  have hcop : Nat.Coprime 6 r := by
    refine Nat.coprime_of_dvd ?_
    intro p hp hp6
    intro hpr
    have hpMul : p ∣ 2 * 3 := by
      norm_num
      exact hp6
    rcases hp.dvd_mul.mp hpMul with hp2 | hp3
    · have peq : p = 2 := (Nat.prime_dvd_prime_iff_eq hp (by norm_num)).mp hp2
      subst p
      rw [Nat.dvd_iff_mod_eq_zero] at hpr
      omega
    · have peq : p = 3 := (Nat.prime_dvd_prime_iff_eq hp (by norm_num)).mp hp3
      subst p
      rw [Nat.dvd_iff_mod_eq_zero] at hpr
      omega
  have h6ge : 4 ≤ ArithmeticFunction.sigma 0 6 := by
    rw [ArithmeticFunction.sigma_zero_apply]
    have hs : ({1, 2, 3, 6} : Finset ℕ) ⊆ (6 : ℕ).divisors := by
      intro x hx
      simp only [Finset.mem_insert, Finset.mem_singleton] at hx
      rw [Nat.mem_divisors]
      constructor
      · rcases hx with rfl | rfl | rfl | rfl <;> norm_num
      · norm_num
    norm_num at hs ⊢
    exact Finset.card_le_card hs
  have comp : ¬ Nat.Prime r → 3 ≤ ArithmeticFunction.sigma 0 r := by
    intro hnp
    rcases (Nat.not_prime_iff_exists_dvd_ne hr2).mp hnp with ⟨d, hd, hd1, hdr⟩
    rw [ArithmeticFunction.sigma_zero_apply]
    have hs : ({1, d, r} : Finset ℕ) ⊆ r.divisors := by
      intro x hx
      simp only [Finset.mem_insert, Finset.mem_singleton] at hx
      rw [Nat.mem_divisors]
      constructor
      · rcases hx with rfl | rfl | rfl
        · exact one_dvd r
        · exact hd
        · exact dvd_rfl
      · omega
    have h1d : 1 ≠ d := Ne.symm hd1
    have h1r : 1 ≠ r := by omega
    have hcard : ({1, d, r} : Finset ℕ).card = 3 := by
      simp [h1d, hd1, hdr, h1r]
    calc
      3 = ({1, d, r} : Finset ℕ).card := hcard.symm
      _ ≤ r.divisors.card := Finset.card_le_card hs
  by_contra hprime
  have hcomp := comp hprime
  have hmul : ArithmeticFunction.sigma 0 (6 * r) =
      ArithmeticFunction.sigma 0 6 * ArithmeticFunction.sigma 0 r :=
    ArithmeticFunction.isMultiplicative_sigma.map_mul_of_coprime hcop
  have hprod :
      12 ≤ ArithmeticFunction.sigma 0 6 * ArithmeticFunction.sigma 0 r := by
    calc
      12 = 4 * 3 := by norm_num
      _ ≤ ArithmeticFunction.sigma 0 6 * ArithmeticFunction.sigma 0 r :=
        Nat.mul_le_mul h6ge hcomp
  have hlarge : 12 ≤ ArithmeticFunction.sigma 0 (6 * r) := by
    rw [hmul]
    exact hprod
  rw [hr] at shift
  omega
