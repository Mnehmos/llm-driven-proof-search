          exact Finset.one_le_prod (fun y _ => by omega)

          simpa [one_mul] using Nat.mul_le_mul_right (a x) hprod

          1 + (a x + s.sum a) = a x + (1 + s.sum a) := by omega

                s.prod (fun y => a y + 1) := Nat.add_le_add_right hmul _
