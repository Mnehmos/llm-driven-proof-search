intro m
induction m using Nat.strong_induction_on with
| _ m ih =>

    native_decide

      push_neg at h0

      · rw [h0, mul_zero] at heq

      · exact h0

      have := Nat.le_self_pow (by omega : a ≠ 0) P
      omega
