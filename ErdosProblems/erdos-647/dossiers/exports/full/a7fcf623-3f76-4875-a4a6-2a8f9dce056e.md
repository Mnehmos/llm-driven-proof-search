theorem root_theorem : ∀ r : ℕ, 2 ≤ r → ArithmeticFunction.sigma 0 r ≤ 3 → Nat.Prime r ∨ ∃ p : ℕ, Nat.Prime p ∧ r = p ^ 2 := by

  by_cases hprime : Nat.Prime r
  · exact Or.inl hprime
  · right
    let p := Nat.minFac r
    let s := r / p
    have hr0 : r ≠ 0 := by omega
    have hr1 : r ≠ 1 := by omega
    have hpprime : Nat.Prime p := by
      dsimp [p]
      exact Nat.minFac_prime hr1
    have hp2 : 2 ≤ p := hpprime.two_le
    have hpdiv : p ∣ r := by
      dsimp [p]
      exact Nat.minFac_dvd r
    have hmul : p * s = r := by
      dsimp [s]
      exact Nat.mul_div_cancel' hpdiv
    have hle : p ≤ s := by
      dsimp [p, s]
      exact Nat.minFac_le_div (by omega) hprime
    by_cases heq : p = s
    · refine ⟨p, hpprime, ?_⟩
      simpa [heq, pow_two] using hmul.symm
    · have hps : p < s := lt_of_le_of_ne hle heq
      have hs2 : 2 ≤ s := le_trans hp2 hle
      have hsltprod : s < p * s :=
        lt_mul_of_one_lt_left (by omega) (by omega)
      have hslt : s < r := by simpa [hmul] using hsltprod
      have hpr : p < r := lt_trans hps hslt
      have h1p : 1 ≠ p := by omega
      have h1s : 1 ≠ s := by omega
      have h1r : 1 ≠ r := by omega
      have hpsne : p ≠ s := by omega
      have hprne : p ≠ r := by omega
      have hsrne : s ≠ r := by omega
      have hcard : ({1, p, s, r} : Finset ℕ).card = 4 := by
        simp [h1p, h1s, h1r, hpsne, hprne, hsrne]
      have hsub : ({1, p, s, r} : Finset ℕ) ⊆ r.divisors := by
        intro x hx
        simp only [Finset.mem_insert, Finset.mem_singleton] at hx
        rcases hx with hx | hx | hx | hx
        · subst x
          exact Nat.one_mem_divisors.mpr hr0
        · subst x
          exact Nat.mem_divisors.mpr ⟨hpdiv, hr0⟩
        · subst x
          apply Nat.mem_divisors.mpr
          constructor
          · refine ⟨p, ?_⟩
            simpa [Nat.mul_comm] using hmul.symm
          · exact hr0
        · subst x
          exact Nat.mem_divisors_self r hr0
      have hcard_lower : 4 ≤ r.divisors.card := by

        rw [hcard] at hc
        exact hc
      have hsigma_card :

        rw [ArithmeticFunction.sigma_apply]
